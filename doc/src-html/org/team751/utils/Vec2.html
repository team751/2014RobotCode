001package org.team751.utils;
002
003import com.sun.squawk.util.MathUtils;
004
005/**
006 * A vector with two elements of type double.
007 * 
008 * All modifying methods return a newly allocated instance, rather than modifying
009 * the current one.
010 * 
011 * @author Sam Crow
012 */
013public class Vec2 {
014    
015    /**
016     * The X (horizontal) component
017     */
018    protected double x = 0;
019    
020    /**
021     * The Y (vertical) component
022     */
023    protected double y = 0;
024    
025    public Vec2() {
026        this(0, 0);
027    }
028    
029    public Vec2(double x, double y) {
030        this.x = x;
031        this.y = y;
032    }
033    
034    /**
035     * Get the X component
036     * @return the X component
037     */
038    public double getX() {
039        return x;
040    }
041    
042    /**
043     * Get the Y component
044     * @return the Y component
045     */
046    public double getY() {
047        return y;
048    }
049    
050    /**
051     * Create and return a newly allocated vector with the same values as this one
052     * @return the new vector
053     */
054    public Object clone() {
055        return new Vec2(x, y);
056    }
057    
058    /**
059     * Create a new vector from the sum of this vector and another
060     * @param other The vector to add.
061     * @return the new vector
062     */
063    public Vec2 add(Vec2 other) {
064        return new Vec2(x + other.x, y + other.y);
065    }
066    
067    /**
068     * Create a new vector from the product of this vector and a scalar
069     * @param d The number to multiply by
070     * @return the new vector
071     */
072    public Vec2 multiply(double d) {
073        return new Vec2(x * d, y * d);
074    }
075    
076    /**
077     * Create a new vector with X and Y values that are the negatives of the
078     * values in this vector
079     * @return the new vector
080     */
081    public Vec2 invert() {
082        return new Vec2(-x, -y);
083    }
084    
085    /**
086     * Get the magnitude (hypotenuse length) of this vector
087     * @return the magnitude
088     */
089    public double getMagnitude() {
090        return Math.sqrt((x * x) + (y * y));
091    }
092    
093    /**
094     * Get the angle of this vector
095     * @return The angle of this vector, in degrees, counterclockwise of the
096     * positive X axis
097     */
098    public double getAngleDegrees() {
099        return Math.toDegrees(MathUtils.atan2(y, x));
100    }
101    
102    /**
103     * Create a new vector with a given angle (counterclockwise from the positive
104     * X axis) and magnitude
105     * @param angleDegrees The angle in degrees, counterclockwise from the
106     * positive X axis, that the vector should point
107     * @param magnitude The magnitude of the vector
108     * @return The new vector
109     */
110    public static Vec2 fromAngle(double angleDegrees, double magnitude) {
111        
112        double radians = Math.toRadians(angleDegrees);
113        double newX = magnitude * Math.cos(radians);
114        double newY = magnitude * Math.sin(radians);
115        
116        return new Vec2(newX, newY);
117    }
118}
