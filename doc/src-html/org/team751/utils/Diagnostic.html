007package org.team751.utils;

009import org.team751.vision.VisionNetworkTableCommunication;
010
011/**
012 *
013 * @author sambaumgarten
014 */
015public class Diagnostic {
016    public static final String ANSI_RESET = "\u001B[0m";
017    public static final String ANSI_BLACK = "\u001B[30m";
018    public static final String ANSI_RED = "\u001B[31m";
019    public static final String ANSI_GREEN = "\u001B[32m";
020    public static final String ANSI_YELLOW = "\u001B[33m";
021    public static final String ANSI_BLUE = "\u001B[34m";
022    public static final String ANSI_PURPLE = "\u001B[35m";
023    public static final String ANSI_CYAN = "\u001B[36m";
024    public static final String ANSI_WHITE = "\u001B[37m";
025    
026    public static void runAllDiagnosticTests() {
027        boolean visionPass = runVisionNetworkTablesDiagnostic();
028    }
029
030    public static boolean runVisionNetworkTablesDiagnostic() {
031        System.out.println(ANSI_BLUE + "STARTING VISION NET TABLES DIAGNOSTIC");
032        System.out.println(ANSI_BLUE + "VisionNetworkTableCommunication Initialized");
033        VisionNetworkTableCommunication vntc = new VisionNetworkTableCommunication();
034        System.out.println(ANSI_PURPLE + "CHECK if the nettables connection was established successfully");
035        if (!vntc.isConnectionAvailable()) {
036            System.out.println(ANSI_RED + "FAILED VISION NET TABLES DIAGNOSTIC");
037            return false;
038        } else {
039            System.out.println(ANSI_GREEN + "PASSED VISION NET TABLES DIAGNOSTIC");
040            return true;
041        }
042    }
043}
