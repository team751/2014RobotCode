001package org.team751.utils;
002
003import edu.wpi.first.wpilibj.Utility;
004
005/**
006 * Calculates the near-instantaneous rate of change of a quantity over time
007 * @author Sam Crow
008 */
009public class Differentiator {
010        
011        private double lastValue = 0;
012        
013        /**
014         * The time, in microseconds, at which getDerivative() was last called
015         */
016        private long lastRunTime = Utility.getFPGATime();
017        
018        /**
019         * Get the derivative of the data, in units per second, since this method
020         * was last called
021         * @param currentValue The current value of the data
022         * @return The d/dt of the data
023         */
024        public double getDerivative(double currentValue) {
025                long now = Utility.getFPGATime();
026                double dv = currentValue - lastValue;
027                
028                //Calculate the change in time and convert it into seconds
029                double dt = (now - lastRunTime) / 1000000.0;
030                
031                lastValue = currentValue;
032                lastRunTime = now;
033                return dv / dt;
034        }
035}
