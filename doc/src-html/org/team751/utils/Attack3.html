001package org.team751.utils;
002
003import edu.wpi.first.wpilibj.Joystick;
004import edu.wpi.first.wpilibj.buttons.Button;
005import edu.wpi.first.wpilibj.buttons.JoystickButton;
006
007/**
008 * Represents a Logitech Attack 3 joystick and provides access to its buttons.
009 * <p>
010 * When using the command-based robot system, the OI class would normally
011 * contain something like this:
012 * <p>
013 * <pre>
014 * public class OI {
015 *      
016 *      public final Joystick driveStick = new Joystick(1);
017 * 
018 *      private Button driveTrigger = new JoystickButton(driveStick, 1);
019 *      ...
020 * 
021 *      public OI() {
022 *          driveTrigger.whenPressed(...);
023 *      }
024 * }
025 * </pre>
026 * <p>
027 * This technique could be considered to make the OI class needlessly verbose
028 * because it specifies the mapping between button numbers and descriptions
029 * for each joystick individually.
030 * </p>
031 * <p>
032 * With this class, it is possible to construct an Attack3 and access its buttons
033 * directly.
034 * </p>
035 * 
036 * @author Sam Crow
037 */
038public class Attack3 extends Joystick {
039
040    public Attack3(int port) {
041        super(port);
042    }
043    
044    /**
045     * The trigger on this joystick.
046     */
047    public final Button trigger = new JoystickButton(this, 1);
048    /**
049     * The left side button on the top of the joystick,
050     * button 4.
051     */
052    public final Button topLeft = new JoystickButton(this, 4);
053    /**
054     * The right side button on the top of the joystick,
055     * button 5.
056     */
057    public final Button topRight = new JoystickButton(this, 5);
058    /**
059     * Rearward button on the top of the joystick, button 2
060     */
061    public final Button topBack = new JoystickButton(this, 2);
062    
063    /**
064     * Forward button on the top of the joystick, button 2
065     */
066    public final Button topFront = new JoystickButton(this, 3);
067    /**
068     * Forward button on the left side of the base, button 6
069     */
070    public final Button baseLeftForward = new JoystickButton(this, 6);
071    /**
072     * Rearward button on the left side of the base, button 7
073     */
074    public final Button baseLeftBack = new JoystickButton(this, 7);
075    
076    /**
077     * Forward button on the right side of the base, button 11
078     */
079    public final Button baseRightForward = new JoystickButton(this, 11);
080    /**
081     * Rearward button on the right side of the base, button 10
082     */
083    public final Button baseRightBack = new JoystickButton(this, 10);
084        /**
085         * Pair of buttons on the back of the joystick, left one of the pair
086         */
087        public final Button baseRearLeft = new JoystickButton(this, 11);
088        /**
089         * Pair of buttons on the back of the joystick, right one of the pair
090         */
091        public final Button baseRearRight = new JoystickButton(this, 12);
092}
