001package org.team751.commands;
002
003import edu.wpi.first.wpilibj.command.Command;
004import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
005import org.team751.base.OI;
006import org.team751.base.RobotMap;
007import org.team751.subsystems.Drivetrain;
008import org.team751.utils.Navigator;
009
010/**
011 * The base for all commands. All atomic commands should subclass CommandBase.
012 * CommandBase stores creates and stores each control system. To access a
013 * subsystem elsewhere in your code in your code use CommandBase.exampleSubsystem
014 * @author sambaumgarten
015 */
016public abstract class CommandBase extends Command {
017
018    public static OI oi;
019    // Create a single static instance of all of your subsystems
020    public static Drivetrain driveTrain = new Drivetrain();
021    //Periodic tasks here (these are not subsystems)
022    public static Navigator navigator = new Navigator();
023
024    public static void init() {
025        // This MUST be here. If the OI creates Commands (which it very likely
026        // will), constructing it during the construction of CommandBase (from
027        // which commands extend), subsystems are not guaranteed to be
028        // yet. Thus, their requires() statements may grab null pointers. Bad
029        // news. Don't move it.
030        oi = new OI();
031        
032        RobotMap.leftDriveEncoder.start();
033        RobotMap.rightDriveEncoder.start();
034        RobotMap.leftDriveEncoder.setDistancePerPulse(.000623);
035        RobotMap.rightDriveEncoder.setDistancePerPulse(.000623);
036
037        // Show what command your subsystem is running on the SmartDashboard
038        SmartDashboard.putData(driveTrain);
039    }
040
041    public CommandBase(String name) {
042        super(name);
043    }
044
045    public CommandBase() {
046        super();
047    }
048}
