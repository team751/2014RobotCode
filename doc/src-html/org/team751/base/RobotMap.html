007/**
008 * The RobotMap is a mapping from the ports sensors and actuators are wired into
009 * to a variable name. This provides flexibility changing wiring, makes checking
010 * the wiring easier and significantly reduces the number of magic numbers
011 * floating around.
012 */
013public class RobotMap {
014    // For example to map the left and right motors, you could define the
015    // following variables to use with your drivetrain subsystem.
016    // public static final int leftMotor = 1;
017    // public static final int rightMotor = 2;
018    
019    // If you are using multiple modules, make sure to define both the port
020    // number and the module. For example you with a rangefinder:
021    // public static final int rangefinderPort = 1;
022    // public static final int rangefinderModule = 1;
023    
024    // Connections
025    public static final int leftDrivePWM = 1;
026    public static final int rightDrivePWM = 2;
027    public static final int leftDriveEncoderChannelA = 1;
028    public static final int leftDriveEncoderChannelB = 2;
029    public static final int rightDriveEncoderChannelA = 3;
030    public static final int rightDriveEncoderChannelB = 4;
031    public static final int gyroChannel = 1;
032    public static final int ultrasonicPingChannel = 2;
033    public static final int ultrasonicEchoChannel = 3;
034    
035    // Speed Controllers
036    public static final Jaguar leftDrivetrainJaguar = new Jaguar(RobotMap.leftDrivePWM);
037    public static final Jaguar rightDrivetrainJaguar = new Jaguar(RobotMap.rightDrivePWM);
038    
039    // Encoders
040    public static final Encoder leftDriveEncoder = new Encoder(RobotMap.leftDriveEncoderChannelA, RobotMap.leftDriveEncoderChannelB);
041    public static final Encoder rightDriveEncoder = new Encoder(RobotMap.rightDriveEncoderChannelA, RobotMap.rightDriveEncoderChannelB);
