004import org.team751.commands.Autonomous;

009/**
010 * This class is the glue that binds the controls on the physical operator
011 * interface to the commands and command groups that allow control of the robot.
012 */
013public class OI {
014    
015    public Attack3 leftJoystick = new Attack3(1);
016    public Attack3 rightJoystick = new Attack3(2);
017    
018    public OI() {
019        // Drive forwards
020        leftJoystick.baseRearLeft.whenPressed(new DriveStraight(5*0.3048));
021        leftJoystick.baseLeftForward.whileHeld(new HoldAngle(10));
022    }
023    
024    //// CREATING BUTTONS
025    // One type of button is a joystick button which is any button on a joystick.
026    // You create one by telling it which joystick it's on and which button
027    // number it is.
028    // Joystick stick = new Joystick(port);
029    // Button button = new JoystickButton(stick, buttonNumber);
030    
031    // Another type of button you can create is a DigitalIOButton, which is
032    // a button or switch hooked up to the cypress module. These are useful if
033    // you want to build a customized operator interface.
034    // Button button = new DigitalIOButton(1);
035    
036    // There are a few additional built in buttons you can use. Additionally,
037    // by subclassing Button you can create custom triggers and bind those to
038    // commands the same as any other Button.
039    
040    //// TRIGGERING COMMANDS WITH BUTTONS
041    // Once you have a button, it's trivial to bind it to a button in one of
042    // three ways:
043    
044    // Start the command when the button is pressed and let it run the command
045    // until it is finished as determined by it's isFinished method.
046    // button.whenPressed(new ExampleCommand());
047    
048    // Run the command while the button is being held down and interrupt it once
049    // the button is released.
050    // button.whileHeld(new ExampleCommand());
051    
052    // Start the command when the button is released  and let it run the command
053    // until it is finished as determined by it's isFinished method.
054    // button.whenReleased(new ExampleCommand());
055}
