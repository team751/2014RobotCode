065    private static double getDistanceToGoalFromNetworkTables() {
066        NumberArray rectanglesArray = vntc.getRectangles();
067            if (VisionConstants.DEBUG_LEVEL >= 3)
068                System.out.println("Number of points in array: " + rectanglesArray.size());
069            
070            TargetReport target = new TargetReport();
071            if (rectanglesArray.size() > 0) {
072                if (rectanglesArray.size() % 8 != 0) {
073                    System.out.println("Error: Number of points in array not divisible by 8!");
074                }
075                else {
076                    int verticalTargets[] = new int[VisionConstants.MAX_PARTICLES];
077                    int horizontalTargets[] = new int[VisionConstants.MAX_PARTICLES];
078                    int verticalTargetCount, horizontalTargetCount;
079                    
080                    //iterate through each particle and score to see if it is a target
081                    Rect rectangles[] = VisionScoring.getRectsFromArray(rectanglesArray);
082                    int numParticles = rectangles.length;
083                    Scores scores[] = new Scores[numParticles];
084                    horizontalTargetCount = verticalTargetCount = 0;
085                    
086                    for (int i = 0; i < VisionConstants.MAX_PARTICLES && i < numParticles; i++) {
087                        VisionScoring.populateRectangleSizes(rectangles[i]);
088                        scores[i] = new Scores();
089                        //Score each particle on rectangularity and aspect ratio
090                        // set rectangularity to 1.0 for now since we don't have this image/data from NetTable
091                        scores[i].rectangularity = 100; //scoreRectangularity(bbWidth, bbHeight);
092                        scores[i].aspectRatioVertical = VisionScoring.scoreAspectRatio(rectangles[i], true);
093                        scores[i].aspectRatioHorizontal = VisionScoring.scoreAspectRatio(rectangles[i], false);
094                        if (VisionConstants.DEBUG_LEVEL >= 3)
095                            System.out.println("AspectRatios: " + scores[i].aspectRatioVertical + " " + scores[i].aspectRatioHorizontal);
096                        //Check if the particle is a horizontal target, if not, check if it's a vertical target
097                        if(VisionScoring.scoreCompare(scores[i], false))
098                        {
099                            if (VisionConstants.DEBUG_LEVEL >= 3)
100                                System.out.println("particle: " + i + "is a Horizontal Target centerX: " + rectangles[i].center_mass_x + "centerY: " + rectangles[i].center_mass_y);
101                            horizontalTargets[horizontalTargetCount++] = i; //Add particle to target array and increment count
102                        } else if (VisionScoring.scoreCompare(scores[i], true)) {
103                            if (VisionConstants.DEBUG_LEVEL >= 3)
104                                System.out.println("particle: " + i + "is a Vertical Target centerX: " + rectangles[i].center_mass_x + "centerY: " + rectangles[i].center_mass_y);
105                            verticalTargets[verticalTargetCount++] = i;  //Add particle to target array and increment count
106                        } else {
107                            if (VisionConstants.DEBUG_LEVEL >= 3)
108                                System.out.println("particle: " + i + "is not a Target centerX: " + rectangles[i].center_mass_x + "centerY: " + rectangles[i].center_mass_y);
109                        }
110                            if (VisionConstants.DEBUG_LEVEL >= 3)
111                                System.out.println("rect: " + scores[i].rectangularity + "ARHoriz: " + scores[i].aspectRatioHorizontal);
112                            if (VisionConstants.DEBUG_LEVEL >= 3)
113                                System.out.println("ARVert: " + scores[i].aspectRatioVertical); 
114                        }
115                        //Zero out scores and set verticalIndex to first target in case there are no horizontal targets
116                        target.totalScore = target.leftScore = target.rightScore = target.tapeWidthScore = target.verticalScore = 0;
117                        target.verticalIndex = verticalTargets[0];
118                        for (int i = 0; i < verticalTargetCount; i++)
119                        {
120                                Rect verticalRect = rectangles[verticalTargets[i]];
121                                for (int j = 0; j < horizontalTargetCount; j++)
122                                {
123                                    Rect horizontalRect = rectangles[horizontalTargets[j]];
124                                    double horizWidth, horizHeight, vertWidth, leftScore, rightScore, tapeWidthScore, verticalScore, total;
125                                    //Measure equivalent rectangle sides for use in score calculation
126                                    horizWidth = horizontalRect.rectLong;
127                                    vertWidth = verticalRect.rectShort;
128                                    horizHeight = horizontalRect.rectShort;
129                                    //Determine if the horizontal target is in the expected location to the left of the vertical target
130                                    leftScore = VisionScoring.ratioToScore(1.2*(verticalRect.bbLeft - horizontalRect.center_mass_x)/horizWidth);
131                                    //Determine if the horizontal target is in the expected location to the right of the  vertical target
132                                    rightScore = VisionScoring.ratioToScore(1.2*(horizontalRect.center_mass_x - verticalRect.bbLeft - verticalRect.bbWidth)/horizWidth);
133                                    //Determine if the width of the tape on the two targets appears to be the same
134                                    tapeWidthScore = VisionScoring.ratioToScore(vertWidth/horizHeight);
135                                    //Determine if the vertical location of the horizontal target appears to be correct
136                                    verticalScore = VisionScoring.ratioToScore(1-(verticalRect.bbTop - horizontalRect.center_mass_y)/(4*horizHeight));
137                                    total = leftScore > rightScore ? leftScore:rightScore;
138                                    total += tapeWidthScore + verticalScore;
139                                    //If the target is the best detected so far store the information about it
140                                    if(total > target.totalScore)
141                                    {
142                                            target.horizontalIndex = horizontalTargets[j];
143                                            target.verticalIndex = verticalTargets[i];
144                                            target.totalScore = total;
145                                            target.leftScore = leftScore;
146                                            target.rightScore = rightScore;
147                                            target.tapeWidthScore = tapeWidthScore;
148                                            target.verticalScore = verticalScore;
149                                    }
150                                }
151                                //Determine if the best target is a Hot target
152                                target.Hot = VisionHotness.hotOrNot(target);
153                                target.leftHot = VisionHotness.leftHot(target);
154                                Robot.lastTarget = target;
155                            }
156                            if(verticalTargetCount > 0)
157                            {
158                                    //Information about the target is contained in the "target" structure
159                                    //To get measurement information such as sizes or locations use the
160                                    //horizontal or vertical index to get the particle report as shown below
161                                    target.distance = computeDistance(rectangles[target.verticalIndex]);
162                                    if (VisionConstants.DEBUG_LEVEL >= 2) {
163                                    if(target.Hot)
164                                    {
165                                            System.out.println("Hot target located");
166                                            System.out.println("Distance: " + target.distance);
167                                            System.out.println("LeftHot?: " + target.leftHot);
168                                    } else {
169                                            System.out.println("No hot target present");
170                                            System.out.println("Distance: " + target.distance);
171                                    }
172                                    }
173                                    
174                                    target.angle = VisionAngleCalculations.computeAngle(rectangles[target.verticalIndex]);
175                                    vntc.setVisionDistance(target.distance);
176                                    vntc.setVisionAngle(target.angle);
177                                    if (VisionConstants.DEBUG_LEVEL >= 2) {
178                                        System.out.println("Angle: " + target.angle);
179                                        System.out.println("Lateral Distance: " + Math.tan(target.angle) * target.distance);
180                                    }
181                            }
182                        
183                }
184            }
185            return target.distance;
186    }

188    private static double getDistanceToGoalFromCRIO() {
189        TargetReport target = new TargetReport();
190        int verticalTargets[] = new int[VisionConstants.MAX_PARTICLES];
191        int horizontalTargets[] = new int[VisionConstants.MAX_PARTICLES];
192        int verticalTargetCount, horizontalTargetCount;
193        double distance = -1.0;
194
195        try {
196            /**
197             * Do the image capture with the camera and apply the algorithm described above. This
198             * sample will either get images from the camera or from an image file stored in the top
199             * level directory in the flash memory on the cRIO. The file name in this case is "testImage.jpg"
200             * 
201             */
202            ColorImage image = camera.getImage();     // comment if using stored images
203            //ColorImage image;                           // next 2 lines read image from flash on cRIO
204            //image = new RGBImage("/testImage.jpg");           // get the sample image from the cRIO flash
205            BinaryImage thresholdImage = image.thresholdHSV(105, 137, 230, 255, 133, 183);   // keep only green objects
206            //thresholdImage.write("/threshold.bmp");
207            BinaryImage filteredImage = thresholdImage.particleFilter(cc);           // filter out small particles
208            //filteredImage.write("/filteredImage.bmp");
209            //iterate through each particle and score to see if it is a target
210            Scores scores[] = new Scores[filteredImage.getNumberParticles()];
211            horizontalTargetCount = verticalTargetCount = 0;
212            if (filteredImage.getNumberParticles() > 0) {
213              for (int i = 0; i < VisionConstants.MAX_PARTICLES && i < filteredImage.getNumberParticles(); i++) {
214                ParticleAnalysisReport report = filteredImage.getParticleAnalysisReport(i);
215                scores[i] = new Scores();
216                //Score each particle on rectangularity and aspect ratio
217                scores[i].rectangularity = VisionScoring.scoreRectangularity(report);
218                scores[i].aspectRatioVertical = VisionScoring.scoreAspectRatio(filteredImage, report, i, true);
219                scores[i].aspectRatioHorizontal = VisionScoring.scoreAspectRatio(filteredImage, report, i, false);                      
220                //Check if the particle is a horizontal target, if not, check if it's a vertical target
221                if (VisionScoring.scoreCompare(scores[i], false)) {
222                    System.out.println("particle: " + i + "is a Horizontal Target centerX: " + report.center_mass_x + "centerY: " + report.center_mass_y);
223                    horizontalTargets[horizontalTargetCount++] = i; //Add particle to target array and increment count
224                } else if (VisionScoring.scoreCompare(scores[i], true)) {
225                    System.out.println("particle: " + i + "is a Vertical Target centerX: " + report.center_mass_x + "centerY: " + report.center_mass_y);
226                    verticalTargets[verticalTargetCount++] = i;  //Add particle to target array and increment count
227                } else {
228                    System.out.println("particle: " + i + "is not a Target centerX: " + report.center_mass_x + "centerY: " + report.center_mass_y);
229                }
230    
231                System.out.println("rect: " + scores[i].rectangularity + "ARHoriz: " + scores[i].aspectRatioHorizontal);
232                System.out.println("ARVert: " + scores[i].aspectRatioVertical); 
233              }
234              //Zero out scores and set verticalIndex to first target in case there are no horizontal targets
235              target.totalScore = target.leftScore = target.rightScore = target.tapeWidthScore = target.verticalScore = 0;
236              target.verticalIndex = verticalTargets[0];
237              for (int i = 0; i < verticalTargetCount; i++) {
238                ParticleAnalysisReport verticalReport = filteredImage.getParticleAnalysisReport(verticalTargets[i]);
239                for (int j = 0; j < horizontalTargetCount; j++) {
240                  ParticleAnalysisReport horizontalReport = filteredImage.getParticleAnalysisReport(horizontalTargets[j]);
241                  double horizWidth, horizHeight, vertWidth, leftScore, rightScore, tapeWidthScore, verticalScore, total;
242                  //Measure equivalent rectangle sides for use in score calculation
243                  horizWidth = NIVision.MeasureParticle(filteredImage.image, horizontalTargets[j], false, NIVision.MeasurementType.IMAQ_MT_EQUIVALENT_RECT_LONG_SIDE);
244                  vertWidth = NIVision.MeasureParticle(filteredImage.image, verticalTargets[i], false, NIVision.MeasurementType.IMAQ_MT_EQUIVALENT_RECT_SHORT_SIDE);
245                  horizHeight = NIVision.MeasureParticle(filteredImage.image, horizontalTargets[j], false, NIVision.MeasurementType.IMAQ_MT_EQUIVALENT_RECT_SHORT_SIDE);
246                  //Determine if the horizontal target is in the expected location to the left of the vertical target
247                  leftScore = VisionScoring.ratioToScore(1.2*(verticalReport.boundingRectLeft - horizontalReport.center_mass_x)/horizWidth);
248                  //Determine if the horizontal target is in the expected location to the right of the  vertical target
249                  rightScore = VisionScoring.ratioToScore(1.2*(horizontalReport.center_mass_x - verticalReport.boundingRectLeft - verticalReport.boundingRectWidth)/horizWidth);
250                  //Determine if the width of the tape on the two targets appears to be the same
251                  tapeWidthScore = VisionScoring.ratioToScore(vertWidth/horizHeight);
252                  //Determine if the vertical location of the horizontal target appears to be correct
253                  verticalScore = VisionScoring.ratioToScore(1-(verticalReport.boundingRectTop - horizontalReport.center_mass_y)/(4*horizHeight));
254                  total = leftScore > rightScore ? leftScore:rightScore;
255                  total += tapeWidthScore + verticalScore;
256                  //If the target is the best detected so far store the information about it
257                  if (total > target.totalScore) {
258                    target.horizontalIndex = horizontalTargets[j];
259                    target.verticalIndex = verticalTargets[i];
260                    target.totalScore = total;
261                    target.leftScore = leftScore;
262                    target.rightScore = rightScore;
263                    target.tapeWidthScore = tapeWidthScore;
264                    target.verticalScore = verticalScore;
265                  }
266                }
267                //Determine if the best target is a Hot target
268                target.Hot = VisionHotness.hotOrNot(target);
269              }
270              if (verticalTargetCount > 0) {
271                //Information about the target is contained in the "target" structure
272                //To get measurement information such as sizes or locations use the
273                //horizontal or vertical index to get the particle report as shown below
274                ParticleAnalysisReport distanceReport = filteredImage.getParticleAnalysisReport(target.verticalIndex);
275                distance = computeDistance(filteredImage, distanceReport, target.verticalIndex);
276                if(target.Hot) {
277                        System.out.println("Hot target located");
278                        System.out.println("Distance: " + distance);
279                } else {
280                        System.out.println("No hot target present");
281                        System.out.println("Distance: " + distance);
282                }
283              }
284            }
285            /**
286             * all images in Java must be freed after they are used since they are allocated out
287             * of C data structures. Not calling free() will cause the memory to accumulate over
288             * each pass of this loop.
289             */
290            filteredImage.free();
291            thresholdImage.free();
292            image.free();
293          } catch (AxisCameraException ex) {        // this is needed if the camera.getImage() is called
294            ex.printStackTrace();
295          } catch (NIVisionException ex) {
296            ex.printStackTrace();
297          }
298        return distance;
299    }

301    /**
302     * Computes the estimated distance to a target using the height of the particle in the image. For more information and graphics
303     * showing the math behind this approach see the Vision Processing section of the ScreenStepsLive documentation.
304     * 
305     * @param image The image to use for measuring the particle estimated rectangle
306     * @param report The Particle Analysis Report for the particle
307     * @param outer True if the particle should be treated as an outer target, false to treat it as a center target
308     * @return The estimated distance to the target in Inches.
309     */
310    static double computeDistance(BinaryImage image, ParticleAnalysisReport report, int particleNumber) throws NIVisionException {
311            double rectLong, height;
312            int targetHeight;
313            rectLong = NIVision.MeasureParticle(image.image, particleNumber, false, NIVision.MeasurementType.IMAQ_MT_EQUIVALENT_RECT_LONG_SIDE);
314            //using the smaller of the estimated rectangle long side and the bounding rectangle height results in better performance
315            //on skewed rectangles
316            height = Math.min(report.boundingRectHeight, rectLong);
317            targetHeight = 32;
318            return VisionConstants.Y_IMAGE_RES * targetHeight / (height * 2 * Math.tan(VisionConstants.VIEW_ANGLE*Math.PI/(180*2)));
319    }
320    static double computeDistance(Rect verticalRect) {
321            double height;
322            int targetHeight;
323            //using the smaller of the estimated rectangle long side and the bounding rectangle height results in better performance
324            //on skewed rectangles
325            height = Math.min(verticalRect.bbHeight, verticalRect.rectLong);
326            targetHeight = 32;
327            return VisionConstants.Y_IMAGE_RES * targetHeight / (height * 2 * Math.tan(VisionConstants.VIEW_ANGLE*Math.PI/(180*2))); // actually in inches
328    }
